59class AudioBuffer {
60// AudioBuffer will be allocated in PSRAM, If PSRAM not available or has not enough space AudioBuffer will be
61// allocated in FlashRAM with reduced size
62//
63// m_buffer m_readPtr m_writePtr m_endPtr
64// | |<------dataLength------->|<------ writeSpace ----->|
65// ▼ ▼ ▼ ▼
66// ---------------------------------------------------------------------------------------------------------------
67// | <--m_buffSize--> | <--m_resBuffSize --> |
68// ---------------------------------------------------------------------------------------------------------------
69// |<-----freeSpace------->| |<------freeSpace-------->|
70//
71//
72//
73// if the space between m_readPtr and buffend < m_resBuffSize copy data from the beginning to resBuff
74// so that the mp3/aac/flac frame is always completed
75//
76// m_buffer m_writePtr m_readPtr m_endPtr
77// | |<-------writeSpace------>|<--dataLength-->|
78// ▼ ▼ ▼ ▼
79// ---------------------------------------------------------------------------------------------------------------
80// | <--m_buffSize--> | <--m_resBuffSize --> |
81// ---------------------------------------------------------------------------------------------------------------
82// |<--- ------dataLength-- ------>|<-------freeSpace------->|
83//
84//
85
86public:
87 AudioBuffer(size_t maxBlockSize = 0); // constructor
88 ~AudioBuffer(); // frees the buffer
89 size_t init(); // set default values

285 void trim(char *s) {
286 //fb trim in place
287 char *pe;
288 char *p = s;
289 while ( isspace(*p) ) p++; //left
290 pe = p; //right
291 while ( *pe != '\0' ) pe++;
292 do {
293 pe--;
294 } while ( (pe > p) && isspace(*pe) );
295 if (p == s) {
296 *++pe = '\0';
297 } else { //move
298 while ( p <= pe ) *s++ = *p++;
299 *s = '\0';
300 }
301 }

341 int lastIndexOf(const char* haystack, const char* needle) {
342 //fb
343 int nlen = strlen(needle);
344 if (nlen == 0) return -1;
345 const char *p = haystack - nlen + strlen(haystack);
346 while (p >= haystack) {
347 int i = 0;
348 while (needle[i] == p[i])
349 if (++i == nlen) return p - haystack;
350 p--;
351 }
352 return -1;
353 }

361 int specialIndexOf (uint8_t* base, const char* str, int baselen, bool exact = false){
362 int result = 0; // seek for str in buffer or in header up to baselen, not nullterninated
363 if (strlen(str) > baselen) return -1; // if exact == true seekstr in buffer must have "\0" at the end
364 for (int i = 0; i < baselen - strlen(str); i++){
365 result = i;
366 for (int j = 0; j < strlen(str) + exact; j++){
367 if (*(base + i + j) != *(str + j)){
368 result = -1;
369 break;
370 }
371 }
372 if (result >= 0) break;
373 }
374 return result;
375 }

378 size_t bigEndian(uint8_t* base, uint8_t numBytes, uint8_t shiftLeft = 8){
379 uint64_t result = 0;
380 if(numBytes < 1 || numBytes > 8) return 0;
381 for (int i = 0; i < numBytes; i++) {
382 result += *(base + i) << (numBytes -i - 1) * shiftLeft;
383 }
384 if(result > SIZE_MAX) {log_e("range overflow"); result = 0;} // overflow
385 return (size_t)result;
